CREATE DATABASE IF NOT EXISTS acmeArtsDB;
USE acmeArtsDB;

CREATE TABLE Artist(
ArtistID INT NOT NULL AUTO_INCREMENT, 
Name VARCHAR(255) NOT NULL,
PRIMARY KEY(ArtistID)
);

INSERT INTO Artist(Name)
VALUES 
	('August Renoir'),
	('Michelangelo'),
	('Vincent Van Gogh'),
	('Leonardo da Vinci'),
	('Claude Monet'),
	('Pablo Picasso'),
	('Salvador Dali'),
	('Paul Cezanne');
	
	

CREATE TABLE Style(
StyleID INT NOT NULL AUTO_INCREMENT, 
ArtStyle VARCHAR(255) NOT NULL,
PRIMARY KEY(StyleID)
);

INSERT INTO Style(ArtStyle)
VALUES 
	('Impressionism'),
	('Mannerism'),
	('Still-life'),
	('Portrait'),
	('Realism'),
	('Cubism'),
	('Surrealism');

CREATE TABLE Media(
MediaID INT NOT NULL AUTO_INCREMENT, 
Medium VARCHAR(255) NOT NULL,
PRIMARY KEY(MediaID)
);

INSERT INTO Media(Medium)
VALUES 
	('oil'),
	('pen-ink');


CREATE TABLE Painting(
PaintingID INT NOT NULL AUTO_INCREMENT,
Title VARCHAR(255) NOT NULL,
Year INT NOT NULL,
Thumbnail BLOB,
Image LONGBLOB,
ArtistID INT,
StyleID INT,
MediaID INT,
PRIMARY KEY(PaintingID),
FOREIGN KEY(ArtistID) REFERENCES Artist(ArtistID),
FOREIGN KEY(StyleID) REFERENCES Style(StyleID),
FOREIGN KEY(MediaID) REFERENCES Media(MediaID)
);
